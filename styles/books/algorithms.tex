%%
%% Algorithm Commands for Books
%%
%
%% Insert an algorithm
%%
%% #1: the label; algo: will be pre-pended
%% #2: the caption of the algorithm
%% #3: the algorithm pseudo code
\protected\gdef\insertAlgorithm#1#2#3{%
\begin{figure}[tb]\centering%
\begingroup%
\makeatletter%
\let\@latex@error\@gobble%
\makeatother%
\parbox{0.85\linewidth}{%
\begin{algorithm}[H]%
#3%
\noindent%
\centering%
\caption{#2}%
\label{algo:#1}%
\end{algorithm}%
}%
\endgroup%%
\end{figure}%
}%
%
