%%
%% Some pre-defined texgit format commands.
%
%% The separator between the code and output Python listings.
%% #1 the name of the python script
\protected\gdef\listingSepPython#1{%
$\downarrow$~~\expandafter\inlinelistingbox{\texttt{python3~#1}}~~$\downarrow$%
}%
%
%% The separator between the code and output Bash listings.
%% #1 the name of the python script
\protected\gdef\listingSepBash#1{%
$\downarrow$~~\expandafter\inlinelistingbox{\texttt{bash~#1}}~~$\downarrow$%
}%
%
%% Create a hyperlink based on the program name and program url to the given program.
%% #1 the label, lst: will be prepended
\protected\gdef\programUrl#1{%
\href{\gitUrl{lst:#1}}{\texttt{\gitNameEsc{lst:#1}}}}%
%
%% Get the stdout caption
%% the label
\protected\gdef\captionStdout#1{%
The \pgls{stdout} of the program~\programUrl{#1} given in~\cref{lst:#1}.%
}%
%
%% Get the stdout/stderr/exitcode caption
%% #1 the label
\protected\gdef\captionStdall#1{%
The \pgls{stdout}, \pgls{stderr}, and \pgls{exitCode} of the program~\programUrl{#1} given in~\cref{lst:#1}.%
}%
%
%% Print the argument listing as a syntax listing.
%% #1 the argument command that is used as special syntax listing
\protected\gdef\asSyntax#1{%
\let\@old@lst@visiblespace\lst@visiblespace%
\def\lst@visiblespace{{\color{listing-spaces}\@old@lst@visiblespace}}%
#1%
\let\lst@visiblespace\@old@lst@visiblespace%
}%
%
%% Load the right formatting tools.
\ifIsBook\expandafter\@firstoftwo\else\expandafter\@secondoftwo\fi{%
\input{\bookbaseDir/styles/books/texgit_format.tex}%
}{}%
\ifIsSlides\expandafter\@firstoftwo\else\expandafter\@secondoftwo\fi{%
\input{\bookbaseDir/styles/slides/texgit_format.tex}%
}{}%
