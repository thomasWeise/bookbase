%%
%% Tools for tables.
%%
%
\RequirePackage{booktabs}%
%
%% Begin a table header
\gdef\tableHeader{\rowcolor{table-header}}%
%
%% An environment for nice tables.
\newenvironment{booktabular}[2][2]{%
\renewcommand{\arraystretch}{1.2}%
\rowcolors{#1}{table-row}{}%
\begin{tabular}{#2}\arrayrulecolor{table-rule}\toprule\tableHeader%
}{%
\bottomrule\end{tabular}%
}
