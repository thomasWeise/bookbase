%% The separator between the code and output Python listings.
%% #1 the name of the python script
\protected\gdef\listingSepPython#1{%
$\downarrow$~~\expandafter\bashil{python3 #1}~~$\downarrow$%
}%
%
%% The separator between the code and output Bash listings.
%% #1 the name of the python script
\protected\gdef\listingSepBash#1{%
$\downarrow$~~\expandafter\bashil{bash #1}~~$\downarrow$%
}%
